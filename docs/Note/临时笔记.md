```cpp
// log(x) -> lg[x];
// st[i][0] 代表原始数组
void init(){
    // 1.初始化 lg 数组
    lg[1] = 0;
    for(int i = 2; i <= n; i++) lg[i] = lg[i >> 1] + 1;

    // 2.初始化 st 数组
    for(int i = 1; i <= n; i++) st[i][0] = a[i]; // 第0层
    for(int k = 1; k <= lg[n]; k++){
        for(int i = 1; i + (1 << k) - 1 <= n; i++){
            st[i][k] = max(st[i][k - 1], st[i + (1 << (k - 1))][k - 1]);
        }
    }
}

int query(int l, int r){
    int k = lg[r - l + 1];
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}
```
